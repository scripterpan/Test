local function findValidTree()
    for _, tree in ipairs(treeFolder:GetChildren()) do
        local treePart = tree:FindFirstChild("Neon", true)
            or tree:FindFirstChild("MainPart")
            or tree.PrimaryPart
            or tree:FindFirstChildWhichIsA("BasePart")
        
        if treePart and treePart:IsDescendantOf(workspace) then
            return treePart
        end
    end
    return nil
end

                    root.CFrame = CFrame.new(currentTree.Position)
                else
                    currentTree = nil
                end
            end

local chestPatterns = {
    Default = function(n) return n:find("chest") and not (n:find("huge") or n:find("giant")) end,
    Huge = function(n) return n:find("huge") end,
    Giant = function(n) return n:find("giant") end
}

local function getNearest()
    local folder = Workspace:FindFirstChild("ChestFolder")
    local char = LocalPlayer.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    
    if not (folder and root) then return end
    
    local nearest, minDist
    local wPos = worldPos[config.world].Position
    local pattern = chestPatterns[config.chestType]
    
    for _, chest in ipairs(folder:GetChildren()) do
        local hit = chest:FindFirstChild("Hitpart")
        if hit and pattern(chest.Name:lower()) then
            if (hit.Position - wPos).Magnitude < 500 then
                local dist = (root.Position - hit.Position).Magnitude
                if not minDist or dist < minDist then
                    minDist = dist
                    nearest = chest
                end
            end
        end
    end
    
    return nearest
end

                local chest = getNearest()
                
                if chest then
                    local hit = chest:FindFirstChild("Hitpart")
                    local prompt = hit and hit:FindFirstChildOfClass("ProximityPrompt")
                    
                    if hit and prompt then
                        root.CFrame = hit.CFrame + Vector3.new(0, 3, 0)
                        task.wait(0.1)
                        fireproximityprompt(prompt)
                        task.wait(0.1)
                    end
                else
                    task.wait(1)
                end
            end
        end
    end
end)

local function apply()
    if not Toggles.EnableSizeChange.Value then return end
    
    local char = player.Character
    if not char then return end
    
    local tool = char:FindFirstChildOfClass("Tool")
    if not tool then return end
    
    local handle = tool:FindFirstChild("Handle")
    if not handle then return end
    
    if not cache[handle] then
        cache[handle] = Vector3.new(handle.Size.X, 0, handle.Size.Z)
    end
    
    local xz = cache[handle]
    handle.Size = Vector3.new(xz.X, Options.HandleSize.Value, xz.Z)
end

Options.HandleSize:OnChanged(apply)

Toggles.EnableSizeChange:OnChanged(function(v)
    if conn then conn:Disconnect() conn = nil end
    cache = {}
    
    if v then
        conn = RunService.Heartbeat:Connect(function()
            task.wait(0.3)
            apply()
        end)
        apply()
    end
end)

local chestConfig = {
    ["huge"] = {
        name = "💎 Huge Chest",
        color = Color3.fromRGB(255, 85, 255)
    },
    ["giant"] = {
        name = "⭐ Giant Chest",
        color = Color3.fromRGB(138, 43, 226)
    },
    ["default"] = {
        name = "📦 Chest",
        color = Color3.fromRGB(0, 255, 127)
    }
}

local function getChestConfig(chestName)
    local name = chestName:lower()
    if name:find("huge") then
        return chestConfig["huge"]
    elseif name:find("giant") then
        return chestConfig["giant"]
    else
        return chestConfig["default"]
    end
end

local function createChestESP(chest)
    local hitPart = chest:FindFirstChild("Hitpart")
    if not hitPart then return end

    local config = getChestConfig(chest.Name)
    
    local billboardGui = Instance.new("BillboardGui")
    billboardGui.Name = "ChestMarker"
    billboardGui.Adornee = hitPart
    billboardGui.Size = UDim2.new(0, 200, 0, 50)
    billboardGui.StudsOffset = Vector3.new(0, 3, 0)
    billboardGui.AlwaysOnTop = true
    billboardGui.MaxDistance = 1200
    billboardGui.LightInfluence = 0

    local container = Instance.new("Frame")
    container.Size = UDim2.new(1, 0, 1, 0)
    container.BackgroundTransparency = 1
    container.Parent = billboardGui

    local chestLabel = Instance.new("TextLabel")
    chestLabel.Size = UDim2.new(1, 0, 0.5, 0)
    chestLabel.Position = UDim2.new(0, 0, 0, 0)
    chestLabel.BackgroundTransparency = 1
    chestLabel.BorderSizePixel = 0
    chestLabel.Text = config.name
    chestLabel.Font = Enum.Font.GothamBold
    chestLabel.TextSize = 16
    chestLabel.TextColor3 = config.color
    chestLabel.TextStrokeTransparency = 0.3
    chestLabel.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    chestLabel.Parent = container

    local character = CurrentPlayer.Character
    local rootPart = character and character:FindFirstChild("HumanoidRootPart")
    
    local distanceLabel = Instance.new("TextLabel")
    distanceLabel.Size = UDim2.new(1, 0, 0.5, 0)
    distanceLabel.Position = UDim2.new(0, 0, 0.5, 0)
    distanceLabel.BackgroundTransparency = 1
    distanceLabel.BorderSizePixel = 0
    distanceLabel.Font = Enum.Font.Gotham
    distanceLabel.TextSize = 14
    distanceLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    distanceLabel.TextStrokeTransparency = 0.3
    distanceLabel.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    distanceLabel.Parent = container

    if rootPart then
        local distance = math.floor((rootPart.Position - hitPart.Position).Magnitude)
        distanceLabel.Text = "📍 " .. distance .. "m"
    else
        distanceLabel.Text = "📍 ?m"
    end

    billboardGui.Parent = hitPart

    local connection
    connection = RunService.Heartbeat:Connect(function()
        if not billboardGui or not billboardGui.Parent then
            if connection then
                connection:Disconnect()
            end
            return
        end
        
        local char = CurrentPlayer.Character
        local root = char and char:FindFirstChild("HumanoidRootPart")
        if root then
            local dist = math.floor((root.Position - hitPart.Position).Magnitude)
            distanceLabel.Text = "" .. dist .. "m"
        end
    end)

    return billboardGui
end

local function isValidChestType(chestName)
    local name = chestName:lower()
    if selectedChestKind == "Default" and name:find("chest") and not (name:find("huge") or name:find("giant")) then
        return true
    elseif selectedChestKind == "Huge" and name:find("huge") then
        return true
    elseif selectedChestKind == "Giant" and name:find("giant") then
        return true
    end
    return false
end

local function updateChestESP()
    if not isESPEnabled then return end
    
    local chestFolder = Workspace:FindFirstChild("ChestFolder")
    if not chestFolder then return end

    local character = CurrentPlayer.Character
    local rootPart = character and character:FindFirstChild("HumanoidRootPart")
    if not rootPart then return end

    for _, chest in ipairs(chestFolder:GetChildren()) do
        local hitPart = chest:FindFirstChild("Hitpart")
        
        if hitPart and isValidChestType(chest.Name) then
            local distance = (rootPart.Position - hitPart.Position).Magnitude
            local existingESP = hitPart:FindFirstChild("ChestMarker")
            
            if distance < 300 then
                if not existingESP then
                    createChestESP(chest)
                end
            elseif existingESP then
                existingESP:Destroy()
            end
        end
    end
end

function removeAllESP()
    local chestFolder = Workspace:FindFirstChild("ChestFolder")
    if not chestFolder then return end

    for _, chest in ipairs(chestFolder:GetChildren()) do
        local hitPart = chest:FindFirstChild("Hitpart")
        if hitPart then
            local esp = hitPart:FindFirstChild("ChestMarker")
            if esp then
                esp:Destroy()
            end
        end
    end
end

RunService:BindToRenderStep("ChestMarkerUpdate", Enum.RenderPriority.Camera.Value, function()
    if isESPEnabled then
        updateChestESP()
    end
end)

local NoClipToggle = NoclipBox:AddToggle("NoClipToggle", {

    Default = false,
})

local function SetNoClip(state)
    local player = game.Players.LocalPlayer
    if player.Character then
        for _, part in pairs(player.Character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanCollide = not state
            end
        end
    end
end

Toggles.NoClipToggle:OnChanged(function(state)
    SetNoClip(state)
end)

game.Players.LocalPlayer.CharacterAdded:Connect(function(character)
    if Toggles.NoClipToggle.Value then
        SetNoClip(true)
    end
end)

game:GetService("RunService").Stepped:Connect(function()
    if Toggles.NoClipToggle.Value then
        SetNoClip(true)
    end
end)

function updatePlayerList()
    playerList = {}
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            table.insert(playerList, player.Name)
        end
    end
    return playerList
end
